   message(STATUS "OpenSSL found, enable SSH support")
   add_definitions(-DSSH_ENABLED)
   include_directories(${OPENSSL_INCLUDE_DIR})
   set(optionalLibs ${optionalLibs} ${OPENSSL_LIBRARIES})

   ssh/auth.cpp
   ssh/channel.cpp
   ssh/crc32.cpp
   ssh/kex.cpp
   ssh/packet.cpp
   ssh/socket.cpp
   ssh/transport.cpp)
