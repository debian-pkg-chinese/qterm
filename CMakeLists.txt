project(qterm)
cmake_minimum_required(VERSION 2.4.6 FATAL_ERROR)

find_package(Qt4 REQUIRED)
find_package(KDE4)
add_subdirectory(src)
configure_file(${CMAKE_CURRENT_SOURCE_DIR}/config.h.cmake ${CMAKE_CURRENT_BINARY_DIR}/config.h)
